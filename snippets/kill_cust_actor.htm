Scripted counter and killing of Thing.

A simple example of using ThingCount and Thing_Destroy.

                
#library "acslib.acs"
#include "zcommon.acs"

script 1000 ENTER{
	print(s:"starting kill detect loop...");

	While (ThingCount(T_NONE, 10) > 0) {
		Delay(10);
	}
	Thing_Destroy(11, 1);
}
