Example WAD showing WR script trigger (special #80) and simple counter. Platforms are lowered on certain counts:

                
#include "zcommon.acs"
bool triggered = 0;
int counter = 0;

script 1 (void){
    counter++;
    if(triggered == 0){
        print(s:"trigger");
	triggered++;
    }
    else{
        print(s:"already triggered");
    }
    print(i:counter);
    /* lower plats on each increment: */
    int tagnum = -1;
    switch(counter){
        case 1:
            tagnum=2;
        break;

        case 2:
            tagnum=3;
        break;

        case 3:
            tagnum=4;
        break;

        case 4:
            tagnum=5;
        break;

        default:

        break;
    }
    floor_lowerbyvalue(tagnum,16,56);
}
